Kode Roblox: Click Gold dan Respawn Emas

local gold = script.Parent local clickDetector = gold:FindFirstChild("ClickDetector") local newGold = game:GetService("ReplicatedStorage"):WaitForChild("Emas") local function MakeNewGold() clickDetector:Destroy() local num = math.random(5, 10) wait(num) local goldClone = newGold:Clone() goldClone.Parent = workspace goldClone.Position = gold.Position gold:Destroy() end clickDetector.MouseClick:Connect(function(plr) local playerStats = plr:FindFirstChild("leaderstats") -- Mendapatkan Gold dan Item local playerEmas = playerStats:FindFirstChild("Emas") playerEmas.Value = playerEmas.Value + 1 gold.Anchored = true gold.Transparency = 1 gold.CanCollide = false MakeNewGold() end)
